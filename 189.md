/**
 * @param {number[]} nums
 * @param {number} k
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var rotate = function(nums, k) {
    if(nums.length === 0){
        return nums;
    }
    k = k % nums.length;
    for(let ind = 0; ind < k;ind++){
        nums.unshift(nums.pop());
    }
    return nums;
};

/* Definition for singly-linked list.
* function ListNode(val, next)

* this.next = (next===undefined ? null : next) 
* }
* @param {ListNode} head 
* @return {ListNode}

var middleNode =
  function (head) {
    let h = head;
    let a = [];
    while (h != null) {
      a.push(h);
      h = h.next;
    }
    let m = Math.floor(a.length / 2);
    return a[m];
  };

```javascript
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
    if(prices.length === 1){
        return 0;
    }
    let profit = 0;
    for(let ind = 1; ind < prices.length; ind++){
        let diff = prices[ind] - prices[ind - 1];
        if(diff > 0){
            profit += diff;
        }
    }
    return profit;
};
```

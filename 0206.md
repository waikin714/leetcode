/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
  let curr = head;
  let lastNode = null;
  while(curr != null)
  {
    let oriNext = curr.next;
    curr.next = lastNode;
    lastNode = curr;
    curr = oriNext;
  }
  return lastNode;
};
